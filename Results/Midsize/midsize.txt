        function agreeLocation(){
                var elChkLocationWrap = document.querySelector('#listLocationAgree');
                var elsChkLocation = elChkLocationWrap.querySelectorAll('input[type="checkbox"]');
                var elBtnAgreeLocation = document.querySelector('#btnAgreeLocation');

                function chkStatusLocation() {
                    var _flag = true;
                    var idx = 0;
                    while (idx < elsChkLocation.length) {
                        if (!elsChkLocation[idx].checked){
                            _flag = false;
                            break;
                        }
                        idx++;
                    }

                    if (!!_flag) {
                        elBtnAgreeLocation.disabled = false;
                        elBtnAgreeLocation.classList.remove('disabled');
                        return;
                    }
                    elBtnAgreeLocation.disabled = true;
                        elBtnAgreeLocation.classList.add('disabled');
                }
                chkStatusLocation();

                for (var i = 0; i < elsChkLocation.length; i++){
                    (function(j){
                        elsChkLocation[j].addEventListener('change', function(){
                        	
                        	var elChkWrap = this.closest('.inp-ch');
                        	if (this.checked) {
                        		elChkWrap.classList.add('checked');
                        	} else {
                        		console.log(elChkWrap)
                        		elChkWrap.classList.remove('checked');
                        	}
			                chkStatusLocation();
                        });
                    })(i);
                }

            }
            agreeLocation();
